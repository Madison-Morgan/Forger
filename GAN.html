def preprocess_image(img, NEW_WIDTH=256, NEW_HEIGHT=256):
    loaded_img = load_img(img)
    resized_img = loaded_img.resize((NEW_WIDTH, NEW_HEIGHT))
    new_img = np.array(img_to_array(resized_img))
    return new_img
